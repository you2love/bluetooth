蓝牙核心概念

蓝牙是一种工作在2.4GHz ISM频段的短距离无线通信协议，采用跳频扩频(FHSS)技术实现抗干扰。

📋 实际通讯数据解析

1. 广播数据包 (ADV_IND)

// 原始数据: 02 01 06 05 FF 01 02 03 04 05
02 01 06        //  Flags: LE_GENERAL_DISC_MODE
05 FF           //  制造商数据: 0xFF01 (自定义)
// 设备名称在 SCAN_RSP 中
// 02 09 4D 59 2D 44 45 56  // "MY-DEV"

2. 连接请求 (CONNECT_IND)

// 发起端发送连接参数
connInterval: 36 (45ms)
slaveLatency: 0
supervisionTimeout: 500
peerAddressType: 0 (public)
peerAddress: AA:BB:CC:DD:EE:FF

3. 发现服务 Response

// ATT Read By Group Type Response
Handle: 0x0001 - 0x000F  // 服务范围
UUID:   180F              // 电池服务

Handle: 0x0010 - 0x0018
UUID:   180A              // 设备信息服务

4. 读取电池电量

// Request: Read Request
Opcode:  0x0A
Handle:  0x0003

// Response: Read Response  
Opcode:  0x0B
Value:   0x64 (100%)

5. 订阅通知

// Write Request: 写入CCC描述符
Opcode:  0x12
Handle:  0x0004 (CCC描述符)
Value:   0x0001 (Notification Enable)

// Response: Write Response
Opcode:  0x13

💻 完整代码示例

Android 完整示例

public class BLEActivity extends AppCompatActivity {

    private BluetoothManager bluetoothManager;
    private BluetoothAdapter bluetoothAdapter;
    private BluetoothGatt bluetoothGatt;
    private String deviceAddress = "AA:BB:CC:DD:EE:FF";

    // 电池服务 UUID
    private static final UUID BATTERY_SERVICE_UUID = UUID.fromString("0000180f-0000-1000-8000-00805f9b34fb");
    private static final UUID BATTERY_LEVEL_UUID = UUID.fromString("00002a19-0000-1000-8000-00805f9b34fb");

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_ble);

        // 1. 初始化蓝牙
        bluetoothManager = (BluetoothManager) getSystemService(Context.BLUETOOTH_SERVICE);
        bluetoothAdapter = bluetoothManager.getAdapter();

        // 2. 检查蓝牙是否可用
        if (bluetoothAdapter == null || !bluetoothAdapter.isEnabled()) {
            Intent enableBtIntent = new Intent(BluetoothAdapter.ACTION_REQUEST_ENABLE);
            startActivityForResult(enableBtIntent, 1);
        }
    }

    // ============ 第一步: 扫描设备 ============
    private void startScan() {
        // 扫描带特定服务的设备
        ScanFilter filter = new ScanFilter.Builder()
                .setServiceUuid(BATTERY_SERVICE_UUID)
                .build();

        ScanSettings settings = new ScanSettings.Builder()
                .setScanMode(ScanSettings.SCAN_MODE_LOW_LATENCY)
                .build();

        bluetoothAdapter.getBluetoothLeScanner()
                .startScan(Arrays.asList(filter), settings, scanCallback);
    }

    private ScanCallback scanCallback = new ScanCallback() {
        @Override
        public void onScanResult(int callbackType, ScanResult result) {
            BluetoothDevice device = result.getDevice();
            // 显示设备: device.getName() + " " + device.getAddress()
            // 找到目标设备后停止扫描
            bluetoothAdapter.getBluetoothLeScanner().stopScan(this);
            // 开始连接
            connectToDevice(device);
        }
    };

    // ============ 第二步: 连接设备 ============
    private void connectToDevice(BluetoothDevice device) {
        // 第三个参数 autoConnect: false = 快速连接
        bluetoothGatt = device.connectGatt(this, false, gattCallback);
    }

    private BluetoothGattCallback gattCallback = new BluetoothGattCallback() {

        @Override
        public void onConnectionStateChange(BluetoothGatt gatt, int status, int newState) {
            if (newState == BluetoothProfile.STATE_CONNECTED) {
                Log.d("BLE", "连接成功");
                // 发现服务
                gatt.discoverServices();
            } else if (newState == BluetoothProfile.STATE_DISCONNECTED) {
                Log.d("BLE", "断开连接");
            }
        }

        @Override
        public void onServicesDiscovered(BluetoothGatt gatt, int status) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                Log.d("BLE", "服务发现完成");

                // 获取电池服务
                BluetoothGattService batteryService = gatt.getService(BATTERY_SERVICE_UUID);
                if (batteryService != null) {
                    // 获取电池电量特征值
                    BluetoothGattCharacteristic batteryLevel = 
                            batteryService.getCharacteristic(BATTERY_LEVEL_UUID);

                    // 读取电量
                    gatt.readCharacteristic(batteryLevel);

                    // 订阅通知
                    gatt.setCharacteristicNotification(batteryLevel, true);

                    // 启用CCC描述符通知
                    BluetoothGattDescriptor descriptor = batteryLevel.getDescriptor(
                            UUID.fromString("00002902-0000-1000-8000-00805f9b34fb"));
                    descriptor.setValue(BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE);
                    gatt.writeDescriptor(descriptor);
                }
            }
        }

        @Override
        public void onCharacteristicRead(BluetoothGatt gatt, 
                BluetoothGattCharacteristic characteristic, int status) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                // 读取到的数据
                byte[] data = characteristic.getValue();
                if (data.length > 0) {
                    int batteryLevel = data[0] & 0xFF;
                    Log.d("BLE", "电池电量: " + batteryLevel + "%");
                    // 更新UI
                    runOnUiThread(() -> tvBattery.setText(batteryLevel + "%"));
                }
            }
        }

        @Override
        public void onCharacteristicChanged(BluetoothGatt gatt,
                BluetoothGattCharacteristic characteristic) {
            // 收到通知数据
            byte[] data = characteristic.getValue();
            int batteryLevel = data[0] & 0xFF;
            Log.d("BLE", "电量变化: " + batteryLevel + "%");
            runOnUiThread(() -> tvBattery.setText(batteryLevel + "%"));
        }
    };

    // 写入数据示例
    private void writeData(byte[] data) {
        BluetoothGattService service = bluetoothGatt.getService(CUSTOM_SERVICE_UUID);
        BluetoothGattCharacteristic characteristic = service.getCharacteristic(CUSTOM_CHAR_UUID);
        
        characteristic.setValue(data);
        characteristic.setWriteType(BluetoothGattCharacteristic.WRITE_TYPE_NO_RESPONSE);
        bluetoothGatt.writeCharacteristic(characteristic);
    }

    // 断开连接
    private void disconnect() {
        if (bluetoothGatt != null) {
            bluetoothGatt.disconnect();
            bluetoothGatt.close();
        }
    }
}

iOS Swift 示例

import CoreBluetooth

class BLEManager: NSObject, CBCentralManagerDelegate, CBPeripheralDelegate {
    
    var centralManager: CBCentralManager!
    var peripheral: CBPeripheral!
    var batteryCharacteristic: CBCharacteristic!
    
    // UUID定义
    let batteryServiceUUID = CBUUID(string: "180F")
    let batteryLevelUUID = CBUUID(string: "2A19")
    
    override init() {
        super.init()
        centralManager = CBCentralManager(delegate: self, queue: nil)
    }
    
    // 扫描设备
    func startScan() {
        centralManager.scanForPeripherals(withServices: nil, options: [
            CBCentralManagerScanOptionAllowDuplicatesKey: false
        ])
    }
    
    // 发现设备
    func centralManager(_ central: CBCentralManager, 
                        didDiscover peripheral: CBPeripheral,
                        advertisementData: [String: Any],
                        rssi RSSI: NSNumber) {
        
        // 找到目标设备
        if peripheral.name == "MY-DEV" {
            self.peripheral = peripheral
            peripheral.delegate = self
            centralManager.stopScan()
            // 连接
            centralManager.connect(peripheral, options: nil)
        }
    }
    
    // 连接成功
    func centralManager(_ central: CBCentralManager, 
                        didConnect peripheral: CBPeripheral) {
        print("连接成功")
        // 发现服务
        peripheral.discoverServices([batteryServiceUUID])
    }
    
    // 发现服务
    func peripheral(_ peripheral: CBPeripheral, 
                    didDiscoverServices error: Error?) {
        guard let services = peripheral.services else { return }
        
        for service in services {
            if service.uuid == batteryServiceUUID {
                // 发现特征值
                peripheral.discoverCharacteristics([batteryLevelUUID], for: service)
            }
        }
    }
    
    // 发现特征值
    func peripheral(_ peripheral: CBPeripheral, 
                    didDiscoverCharacteristicsFor service: CBService,
                    error: Error?) {
        guard let characteristics = service.characteristics else { return }
        
        for characteristic in characteristics {
            if characteristic.uuid == batteryLevelUUID {
                self.batteryCharacteristic = characteristic
                
                // 读取数据
                peripheral.readValue(for: characteristic)
                
                // 订阅通知
                peripheral.setNotifyValue(true, for: characteristic)
            }
        }
    }
    
    // 读取数据
    func peripheral(_ peripheral: CBPeripheral, 
                    didUpdateValueFor characteristic: CBCharacteristic,
                    error: Error?) {
        if let data = characteristic.value {
            let batteryLevel = data[0]
            print("电池电量: \(batteryLevel)%")
        }
    }
    
    // 写入数据
    func writeData(_ data: Data) {
        if let characteristic = batteryCharacteristic {
            peripheral.writeValue(data, for: characteristic, 
                                 type: .withoutResponse)
        }
    }
}
