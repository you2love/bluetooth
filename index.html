蓝牙核心概念

蓝牙是一种工作在2.4GHz ISM频段的短距离无线通信协议，采用跳频扩频(FHSS)技术实现抗干扰。

// 原始数据: 02 01 06 05 FF 01 02 03 04 05

// 02 09 4D 59 2D 44 45 56  // "MY-DEV"

2. 连接请求 (CONNECT_IND)

// 发起端发送连接参数
connInterval: 36 (45ms)

peerAddress: AA:BB:CC:DD:EE:FF

// ATT Read By Group Type Response
Handle: 0x0001 - 0x000F  // 服务范围

💻 完整代码示例

Android 完整示例

public class BLEActivity extends AppCompatActivity {

    private BluetoothManager bluetoothManager;
    private BluetoothAdapter bluetoothAdapter;
    private BluetoothGatt bluetoothGatt;
    private String deviceAddress = "AA:BB:CC:DD:EE:FF";

    // 电池服务 UUID
    private static final UUID BATTERY_SERVICE_UUID = UUID.fromString("0000180f-0000-1000-8000-00805f9b34fb");
    private static final UUID BATTERY_LEVEL_UUID = UUID.fromString("00002a19-0000-1000-8000-00805f9b34fb");

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_ble);

        // 1. 初始化蓝牙
        bluetoothManager = (BluetoothManager) getSystemService(Context.BLUETOOTH_SERVICE);
        bluetoothAdapter = bluetoothManager.getAdapter();

        // 2. 检查蓝牙是否可用
        if (bluetoothAdapter == null || !bluetoothAdapter.isEnabled()) {
            Intent enableBtIntent = new Intent(BluetoothAdapter.ACTION_REQUEST_ENABLE);
            startActivityForResult(enableBtIntent, 1);
        }
    }

    // ============ 第一步: 扫描设备 ============
    private void startScan() {
        // 扫描带特定服务的设备
        ScanFilter filter = new ScanFilter.Builder()
                .setServiceUuid(BATTERY_SERVICE_UUID)
                .build();

        ScanSettings settings = new ScanSettings.Builder()
                .setScanMode(ScanSettings.SCAN_MODE_LOW_LATENCY)
                .build();

        bluetoothAdapter.getBluetoothLeScanner()
                .startScan(Arrays.asList(filter), settings, scanCallback);
    }

    private ScanCallback scanCallback = new ScanCallback() {
        @Override
        public void onScanResult(int callbackType, ScanResult result) {
            BluetoothDevice device = result.getDevice();
            // 显示设备: device.getName() + " " + device.getAddress()
            // 找到目标设备后停止扫描
            bluetoothAdapter.getBluetoothLeScanner().stopScan(this);
            // 开始连接
            connectToDevice(device);
        }
    };

    // ============ 第二步: 连接设备 ============
    private void connectToDevice(BluetoothDevice device) {
        // 第三个参数 autoConnect: false = 快速连接
        bluetoothGatt = device.connectGatt(this, false, gattCallback);
    }

    private BluetoothGattCallback gattCallback = new BluetoothGattCallback() {

        @Override
        public void onConnectionStateChange(BluetoothGatt gatt, int status, int newState) {
            if (newState == BluetoothProfile.STATE_CONNECTED) {
                Log.d("BLE", "连接成功");
                // 发现服务
                gatt.discoverServices();
            } else if (newState == BluetoothProfile.STATE_DISCONNECTED) {
                Log.d("BLE", "断开连接");
            }
        }

        @Override
        public void onServicesDiscovered(BluetoothGatt gatt, int status) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                Log.d("BLE", "服务发现完成");

                // 获取电池服务
                BluetoothGattService batteryService = gatt.getService(BATTERY_SERVICE_UUID);
                if (batteryService != null) {
                    // 获取电池电量特征值
                    BluetoothGattCharacteristic batteryLevel = 
                            batteryService.getCharacteristic(BATTERY_LEVEL_UUID);

                    // 读取电量
                    gatt.readCharacteristic(batteryLevel);

                    // 订阅通知
                    gatt.setCharacteristicNotification(batteryLevel, true);

                    // 启用CCC描述符通知
                    BluetoothGattDescriptor descriptor = batteryLevel.getDescriptor(
                            UUID.fromString("00002902-0000-1000-8000-00805f9b34fb"));
                    descriptor.setValue(BluetoothGattDescriptor.ENABLE_NOTIFICATION_VALUE);
                    gatt.writeDescriptor(descriptor);
                }
            }
        }

        @Override
        public void onCharacteristicRead(BluetoothGatt gatt, 
                BluetoothGattCharacteristic characteristic, int status) {
            if (status == BluetoothGatt.GATT_SUCCESS) {
                // 读取到的数据
                byte[] data = characteristic.getValue();
                if (data.length > 0) {
                    int batteryLevel = data[0] & 0xFF;
                    Log.d("BLE", "电池电量: " + batteryLevel + "%");
                    // 更新UI
                    runOnUiThread(() -> tvBattery.setText(batteryLevel + "%"));
                }
            }
        }

        @Override
        public void onCharacteristicChanged(BluetoothGatt gatt,
                BluetoothGattCharacteristic characteristic) {
            // 收到通知数据
            byte[] data = characteristic.getValue();
            int batteryLevel = data[0] & 0xFF;
            Log.d("BLE", "电量变化: " + batteryLevel + "%");
            runOnUiThread(() -> tvBattery.setText(batteryLevel + "%"));
        }
    };

    // 写入数据示例
    private void writeData(byte[] data) {
        BluetoothGattService service = bluetoothGatt.getService(CUSTOM_SERVICE_UUID);
        BluetoothGattCharacteristic characteristic = service.getCharacteristic(CUSTOM_CHAR_UUID);
        
        characteristic.setValue(data);
        characteristic.setWriteType(BluetoothGattCharacteristic.WRITE_TYPE_NO_RESPONSE);
        bluetoothGatt.writeCharacteristic(characteristic);
    }

    // 断开连接
    private void disconnect() {
        if (bluetoothGatt != null) {
            bluetoothGatt.disconnect();
            bluetoothGatt.close();
        }
    }
}

iOS Swift 示例

import CoreBluetooth

class BLEManager: NSObject, CBCentralManagerDelegate, CBPeripheralDelegate {
    
    var centralManager: CBCentralManager!
    var peripheral: CBPeripheral!
    var batteryCharacteristic: CBCharacteristic!
    
    // UUID定义
    let batteryServiceUUID = CBUUID(string: "180F")
    let batteryLevelUUID = CBUUID(string: "2A19")
    
    override init() {
        super.init()
        centralManager = CBCentralManager(delegate: self, queue: nil)
    }
    
    // 扫描设备
    func startScan() {
        centralManager.scanForPeripherals(withServices: nil, options: [
            CBCentralManagerScanOptionAllowDuplicatesKey: false
        ])
    }
    
    // 发现设备
    func centralManager(_ central: CBCentralManager, 
                        didDiscover peripheral: CBPeripheral,
                        advertisementData: [String: Any],
                        rssi RSSI: NSNumber) {
        
        // 找到目标设备
        if peripheral.name == "MY-DEV" {
            self.peripheral = peripheral
            peripheral.delegate = self
            centralManager.stopScan()
            // 连接
            centralManager.connect(peripheral, options: nil)
        }
    }
    
    // 连接成功
    func centralManager(_ central: CBCentralManager, 
                        didConnect peripheral: CBPeripheral) {
        print("连接成功")
        // 发现服务
        peripheral.discoverServices([batteryServiceUUID])
    }
    
    // 发现服务
    func peripheral(_ peripheral: CBPeripheral, 
                    didDiscoverServices error: Error?) {
        guard let services = peripheral.services else { return }
        
        for service in services {
            if service.uuid == batteryServiceUUID {
                // 发现特征值
                peripheral.discoverCharacteristics([batteryLevelUUID], for: service)
            }
        }
    }
    
    // 发现特征值
    func peripheral(_ peripheral: CBPeripheral, 
                    didDiscoverCharacteristicsFor service: CBService,
                    error: Error?) {
        guard let characteristics = service.characteristics else { return }
        
        for characteristic in characteristics {
            if characteristic.uuid == batteryLevelUUID {
                self.batteryCharacteristic = characteristic
                
                // 读取数据
                peripheral.readValue(for: characteristic)
                
                // 订阅通知
                peripheral.setNotifyValue(true, for: characteristic)
            }
        }
    }
    
    // 读取数据
    func peripheral(_ peripheral: CBPeripheral, 
                    didUpdateValueFor characteristic: CBCharacteristic,
                    error: Error?) {
        if let data = characteristic.value {
            let batteryLevel = data[0]
            print("电池电量: \(batteryLevel)%")
        }
    }
    
    // 写入数据
    func writeData(_ data: Data) {
        if let characteristic = batteryCharacteristic {
            peripheral.writeValue(data, for: characteristic, 
                                 type: .withoutResponse)
        }
    }
}

📁 案例1: 蓝牙传输文件 (OBEX/FTP)

经典蓝牙使用OBEX协议传输文件，BLE使用自定义GATT服务。

经典蓝牙 - OBEX文件传输

// Android 经典蓝牙文件传输
import android.bluetooth.BluetoothDevice;
import android.bluetooth.BluetoothObexTransport;
import javax.obex.HeaderSet;
import javax.obex.ObexSession;
import javax.obex.ClientSession;
import javax.obex.Operation;

public class FileTransfer {

    // 通过OBEX协议发送文件
    public void sendFile(BluetoothDevice device, String filePath) throws IOException {
        BluetoothSocket socket = device.createRfcommSocketToServiceRecord(
            // OBEX UUID
            UUID.fromString("00001106-0000-1000-8000-00805f9b34fb"));

        socket.connect();

        // 创建OBEX会话
        BluetoothObexTransport transport = new BluetoothObexTransport(socket);
        ClientSession session = new ClientSession(transport);

        // 设置请求头
        HeaderSet request = new HeaderSet();
        request.setHeader(HeaderSet.NAME, "test.txt");
        request.setHeader(HeaderSet.LENGTH, 1024L);

        // 发起PUT请求
        Operation operation = session.put(request);
        OutputStream os = operation.openOutputStream();

        // 写入文件数据
        FileInputStream fis = new FileInputStream(filePath);
        byte[] buffer = new byte[1024];
        int len;
        while ((len = fis.read(buffer)) != -1) {
            os.write(buffer, 0, len);
        }

        os.close();
        operation.close();
        session.close();
    }
}

BLE - 自定义文件传输服务

// BLE 文件传输服务定义
// 服务UUID: 0xFE59 (NUS - Nordic UART Service 变体)
#define FILE_TRANSFER_SERVICE_UUID "0000fe59-0000-1000-8000-00805f9b34fb"

// 特征值
#define FILE_DATA_UUID    "0000ff01-0000-1000-8000-00805f9b34fb"  // Notify+Write
#define FILE_CTRL_UUID   "0000ff02-0000-1000-8000-00805f9b34fb"  // Write
#define FILE_SIZE_UUID   "0000ff03-0000-1000-8000-00805f9b34fb"  // Read

// 控制命令
#define CMD_START   0x01  // 开始传输
#define CMD_DATA    0x02  // 数据包
#define CMD_END     0x03  // 传输完成
#define CMD_CANCEL  0x04  // 取消传输

// 分包发送
private void sendFileByBLE(byte[] fileData) {
    int mtu = 512; // 协商后的MTU
    int packetSize = mtu - 3; // ATT header
    int totalPackets = (fileData.length + packetSize - 1) / packetSize;

    // 1. 发送文件大小
    byte[] sizeHeader = new byte[4];
    ByteBuffer.wrap(sizeHeader).putInt(fileData.length);
    writeControl(CMD_START, sizeHeader);

    // 2. 分包发送数据
    for (int i = 0; i < totalPackets; i++) {
        int offset = i * packetSize;
        int len = Math.min(packetSize, fileData.length - offset);
        byte[] packet = new byte[len + 1];
        packet[0] = CMD_DATA;
        System.arraycopy(fileData, offset, packet, 1, len);
        writeData(packet);
    }

    // 3. 发送完成
    writeControl(CMD_END, new byte[0]);
}

// Android A2DP 音频播放
import android.bluetooth.A2dpSinkProfile;
import android.bluetooth.BluetoothA2dpSink;

public class BluetoothAudio {

    private BluetoothAdapter bluetoothAdapter;
    private BluetoothA2dpSink a2dpSink;

    // 获取已配对的A2DP设备
    public List getA2dpDevices() {
        Set devices = bluetoothAdapter.getBondedDevices();
        List a2dpDevices = new ArrayList<>();

        for (BluetoothDevice device : devices) {
            if (device.getBluetoothClass().getMajorDeviceClass()
                    == BluetoothClass.Device.Major.AUDIO_VIDEO) {
                a2dpDevices.add(device);
            }
        }
        return a2dpDevices;
    }

    // 连接A2DP设备
    public boolean connectA2dp(BluetoothDevice device) {
        try {
            Method connectMethod = BluetoothA2dpSink.class.getMethod(
                "connect", BluetoothDevice.class);
            return (boolean) connectMethod.invoke(a2dpSink, device);
        } catch (Exception e) {
            return false;
        }
    }

📱 案例4: 微信小程序蓝牙控制智能设备

// 微信小程序 - 控制智能灯
Page({
    data: {
        deviceId: '',
        serviceId: '',
        characteristicId: '',
        connected: false,
        lightOn: false
    },

    // 1. 初始化蓝牙
    initBluetooth() {
        wx.openBluetoothAdapter({
            success: () => this.startScan(),
            fail: () => wx.showToast({title: '请开启蓝牙', icon: 'none'})
        })
    },

    // 2. 扫描设备
    startScan() {
        wx.startBluetoothDevicesDiscovery({
            success: () => {
                wx.onBluetoothDeviceFound((res) => {
                    res.devices.forEach(device => {
                        if (device.name && device.name.includes('SmartLight')) {
                            this.setData({ deviceId: device.deviceId });
                            wx.stopDiscovery();
                            this.connectDevice(device.deviceId);
                        }
                    })
                })
            }
        })
    },

    // 3. 连接设备
    connectDevice(deviceId) {
        wx.createBLEConnection({
            deviceId,
            success: () => {
                this.setData({ connected: true });
                this.getServices(deviceId);
            }
        })
    },

    // 4. 获取服务
    getServices(deviceId) {
        wx.getBLEDeviceServices({
            deviceId,
            success: (res) => {
                // 找到自定义服务
                const service = res.services.find(s => 
                    s.isPrimary && s.uuid.includes('FE59'));
                if (service) {
                    this.setData({ serviceId: service.uuid });
                    this.getCharacteristics(deviceId, service.uuid);
                }
            }
        })
    },

    // 5. 获取特征值
    getCharacteristics(deviceId, serviceId) {
        wx.getBLEDeviceCharacteristics({
            deviceId,
            serviceId,
            success: (res) => {
                const char = res.characteristics.find(c =>
                    c.properties.write || c.properties.notify);
                if (char) {
                    this.setData({ characteristicId: char.uuid });
                    // 启用通知
                    if (char.properties.notify) {
                        wx.notifyBLECharacteristicValueChange({
                            deviceId, serviceId,
                            characteristicId: char.uuid,
                            state: true
                        });
                    }
                }
            }
        })
    },

    // 6. 控制灯开关
    toggleLight() {
        const buffer = new ArrayBuffer(1);
        const dataView = new DataView(buffer);
        dataView.setUint8(0, this.data.lightOn ? 0x00 : 0x01);

        wx.writeBLECharacteristicValue({
            deviceId: this.data.deviceId,
            serviceId: this.data.serviceId,
            characteristicId: this.data.characteristicId,
            value: buffer,
            success: () => {
                this.setData({ lightOn: !this.data.lightOn });
            }
        });
    }
})

🌡️ 案例6: 体温计数据读取

// Health Thermometer Service
// Service: 0x1809
// Characteristic: 0x2A1E (Temperature Measurement)

// 数据格式 (5 bytes)
struct {
    uint8_t flags;        // bit0: 0=Celsius, 1=Fahrenheit
    int32_t temperature; // IEEE 11073 FLOAT
} temp_measurement;

// 解析体温数据
private float parseTemperature(byte[] data) {
    int flags = data[0];
    boolean isFahrenheit = (flags & 0x01) == 1;

    // IEEE 11073 32-bit FLOAT
    int mantissa = (data[4] << 24) | (data[3] << 16) | (data[2] << 8) | (data[1]);
    int exponent = data[4] >> 8;

    float temperature = mantissa * Math.pow(10, exponent);

    if (isFahrenheit) {
        temperature = temperature * 9 / 5 + 32;
    }

    return temperature;
}
